SELECT buy.client_id as clientId, SUM(book.price*order_line.quantity) as total
FROM buy JOIN order_line ON buy.id = order_line.order_id
JOIN book ON order_line.book_id = book.id

GROUP BY clientId, buy.id
ORDER BY total DESC
